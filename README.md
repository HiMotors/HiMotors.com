# Locação Carros (HiMotors) - Flask

## instalar
python -m venv venv
source venv/bin/activate  # ou venv\Scripts\activate no Windows
pip install -r requirements.txt

## inicializar banco e dados
python init_db.py

## rodar app
python app.py
# abra http://127.0.0.1:5000
